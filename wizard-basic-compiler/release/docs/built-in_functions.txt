Math module:
	NUMBER SIN(NUMBER)
	NUMBER COS(NUMBER)
	NUMBER TAN(NUMBER)
	NUMBER ATN(NUMBER)
	NUMBER EXP(NUMBER)
	NUMBER LOG(NUMBER)
	NUMBER SQR(NUMBER)
	NUMBER ABS(NUMBER)
	NUMBER INT(NUMBER)

System module:
	NUMBER RND(VOID) - return random number in limits [0; 1]
	NUMBER TIMER(VOID) - return time from start in seconds
	VOID PRINT(NUMBER number)
	VOID PRINT(ARRAY string)
	VOID PRINT(NUMBER file_id, ARRAY byte_sequence) - output to file
	NUMBER OPEN(ARRAY filename, NUMBER mode) - mode can be 0 (read), 1 (write)
		and 2 (append)
	NUMBER INPUT(NUMBER file_id) - read byte from file
	VOID CLOSE(NUMBER file_id)

Graphics module:
	NUMBER SCREEN_WIDTH(VOID)
	NUMBER SCREEN_HEIGHT(VOID)
	VOID POSITION_CAMERA(NUMBER x, NUMBER y, NUMBER z)
	VOID ROTATION_CAMERA(NUMBER x, NUMBER y, NUMBER z)
	VOID FOG(NUMBER mode) - mode can be 0 (off) and 1 (on)
	VOID FOG_COLOR(NUMBER r, NUMBER g, NUMBER b)
	VOID FOG_DEPTH(NUMBER depth) - set depth, starting from, fog is maximum
	NUMBER LOAD_OBJECT(ARRAY filename)
	VOID POSITION_OBJECT(NUMBER x, NUMBER y, NUMBER z)
	VOID ROTATION_OBJECT(NUMBER x, NUMBER y, NUMBER z)
	VOID SCALE_OBJECT(NUMBER x, NUMBER y, NUMBER z)
	VOID PLAY_OBJECT(NUMBER object_id, NUMBER start_frame, NUMBER end_frame,
		NUMBER loop) - loop can be 0 (not loop) and 1 (loop)
	VOID STOP_OBJECT(NUMBER object_id)
	NUMBER KEYSTATE(NUMBER key_code)
	NUMBER MOUSECLICK(VOID) - return union flags of mouse buttons
	NUMBER MOUSEX(VOID)
	NUMBER MOUSEY(VOID)
	VOID POSITION_MOUSE(NUMBER x, NUMBER y)
	VOID SYNC(VOID)

Sound module:
	NUMBER LOAD_SOUND(ARRAY filename)
	VOID PLAY_SOUND(NUMBER sound_id, NUMBER loop) - loop can be 0 (not loop) and
		1 (loop)
	VOID STOP_SOUND(NUMBER sound_id)
	VOID POSITION_SOUND(NUMBER sound_id, NUMBER x, NUMBER y, NUMBER z)
	VOID POSITION_LISTENER(NUMBER x, NUMBER y, NUMBER z)
	VOID ROTATION_LISTENER(NUMBER x, NUMBER y, NUMBER z)
